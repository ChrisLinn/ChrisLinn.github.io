    player-t : type
    payoff-t : type
    action-t : (player-t) -> type

    game-t : conj {
      player-vect : vect-t (player-t, :n)
      nof-players = :n
      utility : (
        strategy-profile : vect-t (sigma-t (player-t, action-t), :n)
      ) -> vect-t (sigma-t (player-t, lambda (_) payoff-t), :n)
    }

    sigma-t : conj {
      car-t : type
      cdr-dep-t : (car-t) -> type
      cdr-t = cdr-dep-t (car)

    }
